Here's a self-contained IEC 61131-3 Ladder Diagram program for controlling a 5-floor elevator:

```
(* Inputs *)
VAR_INPUT
  FloorLS : ARRAY[1..5] OF BOOL; (* Floor limit switches *)
  DoorLS_Top : ARRAY[1..5] OF BOOL; (* Door top limit switches *)
  DoorLS_Bottom : ARRAY[1..5] OF BOOL; (* Door bottom limit switches *)  
  FloorCall : ARRAY[1..5] OF BOOL; (* Floor call buttons *)
  FloorUp : ARRAY[1..5] OF BOOL; (* Floor up buttons *)
  FloorDown : ARRAY[1..5] OF BOOL; (* Floor down buttons *)
  CabinButtons : ARRAY[1..5] OF BOOL; (* Cabin floor buttons *)
END_VAR

(* Outputs *)
VAR_OUTPUT  
  MotorUp : BOOL; (* Elevator motor up *)
  MotorDown : BOOL; (* Elevator motor down *)
  DoorOpen : BOOL; (* Door open command *)
  DoorClose : BOOL; (* Door close command *)
  FloorIndicator : INT := 1; (* Current floor indicator *)
END_VAR

(* Internal variables *)
VAR
  DoorOpenTimer : TON; (* Door open timer *)
  DoorReopenTimer : TON; (* Door reopen timer *)
  Direction : INT := 0; (* 1=Up, -1=Down, 0=Idle *)
  TargetFloor : INT := 1; (* Target floor *)
END_VAR
